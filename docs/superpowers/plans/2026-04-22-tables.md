# Tables Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a generic, group-scoped Tables subsystem (admin-managed reference data with named lookups) plus a single universal `tables.lookup` Temporal activity, exposed through a user-friendly schema-driven UI.

**Architecture:** Two new Prisma models (`Table`, `TableRow`) hold all client-specific data and schema as JSONB. A closed set of `ColumnType` values gives end-to-end type safety; lookups are expressed in the existing `ConditionExpression` DSL (extended with `param.X` and `row.X` namespaces). The frontend renders rows from the table's `ColumnDef[]` and offers lookup templates so admins author predicates via a small form. The graph runner auto-injects `groupId` into every activity input so the lookup activity knows which group to query.

**Tech Stack:** Prisma + PostgreSQL, NestJS (backend), Temporal (worker), React + Mantine + React Query + Vitest (frontend), Jest (backend/temporal), Zod (validation).

**Spec:** `docs/superpowers/specs/2026-04-22-tables-design.md`

```typescript
// apps/backend-services/src/tables/column-validation.ts
import { z } from "zod";
import type { ColumnDef, ColumnType } from "./types";

const KEY_PATTERN = /^[a-zA-Z_][a-zA-Z0-9_]*$/;

export function validateColumnDefs(cols: ColumnDef[]): void {
  const seen = new Set<string>();
  for (const col of cols) {
    if (!KEY_PATTERN.test(col.key)) {
      throw new Error(
        `invalid column key "${col.key}": must match ${KEY_PATTERN}`,
      );
    }
    if (seen.has(col.key)) {
      throw new Error(`duplicate column key "${col.key}"`);
    }
    seen.add(col.key);
    if (col.type === "enum") {
      if (!col.enumValues || col.enumValues.length === 0) {
        throw new Error(`column "${col.key}": enumValues required for type enum`);
      }
    } else if (col.enumValues !== undefined) {
      throw new Error(
        `column "${col.key}": enumValues only allowed for type enum`,
      );
    }
  }
}

function zodForColumn(col: ColumnDef): z.ZodTypeAny {
  let base: z.ZodTypeAny;
  switch (col.type) {
    case "string":
      base = z.string();
      break;
    case "number":
      base = z.number();
      break;
    case "boolean":
      base = z.boolean();
      break;
    case "date":
      base = z.string().regex(/^\d{4}-\d{2}-\d{2}$/);
      break;
    case "datetime":
      base = z.string().datetime({ offset: true });
      break;
    case "enum":
      base = z.enum(col.enumValues as [string, ...string[]]);
      break;
    default: {
      const _exhaustive: never = col.type;
      throw new Error(`unknown column type: ${_exhaustive}`);
    }
  }
  return col.required ? base : base.optional();
}

export function buildRowZodSchema(cols: ColumnDef[]): z.ZodTypeAny {
  const shape: Record<string, z.ZodTypeAny> = {};
  for (const col of cols) {
    shape[col.key] = zodForColumn(col);
  }
  // .strip() drops unknown keys silently (default in Zod)
  return z.object(shape).strip();
}
```

```typescript
// apps/backend-services/src/tables/lookup-validation.ts
import type { ColumnDef, LookupDef, LookupParam } from "./types";
import type {
  ConditionExpression,
  ValueRef,
} from "../workflow/graph-workflow-types";

export function validateLookupDefs(
  lookups: LookupDef[],
  cols: ColumnDef[],
): void {
  const colKeys = new Set(cols.map((c) => c.key));
  const seenNames = new Set<string>();
  for (const lookup of lookups) {
    if (seenNames.has(lookup.name)) {
      throw new Error(`duplicate lookup name "${lookup.name}"`);
    }
    seenNames.add(lookup.name);
    const paramNames = new Set(lookup.params.map((p) => p.name));
    validateExpression(lookup.filter, colKeys, paramNames, lookup.name);
    if (lookup.order) {
      for (const o of lookup.order) {
        if (!colKeys.has(o.field)) {
          throw new Error(
            `lookup "${lookup.name}": unknown column "${o.field}" in order clause`,
          );
        }
      }
    }
  }
}

function validateExpression(
  expr: ConditionExpression,
  colKeys: Set<string>,
  paramNames: Set<string>,
  lookupName: string,
): void {
  // Walk the expression tree and validate every ValueRef.
  switch (expr.operator) {
    case "and":
    case "or":
      for (const sub of expr.expressions) {
        validateExpression(sub, colKeys, paramNames, lookupName);
      }
      return;
    case "not":
      validateExpression(expr.expression, colKeys, paramNames, lookupName);
      return;
    case "is-null":
    case "is-not-null":
      validateRef(expr.value, colKeys, paramNames, lookupName);
      return;
    case "in":
    case "not-in":
      validateRef(expr.value, colKeys, paramNames, lookupName);
      for (const item of expr.list) {
        validateRef(item, colKeys, paramNames, lookupName);
      }
      return;
    default:
      // ComparisonExpression: equals, not-equals, gt, gte, lt, lte, contains
      validateRef(expr.left, colKeys, paramNames, lookupName);
      validateRef(expr.right, colKeys, paramNames, lookupName);
      return;
  }
}

function validateRef(
  ref: ValueRef,
  colKeys: Set<string>,
  paramNames: Set<string>,
  lookupName: string,
): void {
  if ("literal" in ref) return;
  const path = ref.ref;
  if (path.startsWith("row.")) {
    const key = path.slice("row.".length).split(".")[0];
    if (!colKeys.has(key)) {
      throw new Error(
        `lookup "${lookupName}": unknown column "${key}" in row.X reference`,
      );
    }
  } else if (path.startsWith("param.")) {
    const key = path.slice("param.".length).split(".")[0];
    if (!paramNames.has(key)) {
      throw new Error(
        `lookup "${lookupName}": undeclared param "${key}"`,
      );
    }
  }
  // ctx.X references are unrestricted (admin's responsibility)
}
```

> Note: `ConditionExpression` and `ValueRef` types are imported from the existing graph-workflow-types. If their structures don't match the discriminator names used above (e.g., `expressions`, `expression`, `value`, `list`), inspect `apps/backend-services/src/workflow/graph-workflow-types.ts` and adjust the field accesses to match.

```typescript
// apps/backend-services/src/tables/dependency-check.ts
import type { LookupDef } from "./types";
import type { ConditionExpression, ValueRef } from "../workflow/graph-workflow-types";

export function findLookupsReferencingColumn(
  lookups: LookupDef[],
  columnKey: string,
): string[] {
  const result: string[] = [];
  for (const lookup of lookups) {
    if (lookupReferencesColumn(lookup, columnKey)) {
      result.push(lookup.name);
    }
  }
  return result;
}

function lookupReferencesColumn(lookup: LookupDef, key: string): boolean {
  if (lookup.order?.some((o) => o.field === key)) return true;
  return expressionReferencesColumn(lookup.filter, key);
}

function expressionReferencesColumn(
  expr: ConditionExpression,
  key: string,
): boolean {
  switch (expr.operator) {
    case "and":
    case "or":
      return expr.expressions.some((s) => expressionReferencesColumn(s, key));
    case "not":
      return expressionReferencesColumn(expr.expression, key);
    case "is-null":
    case "is-not-null":
      return refReferencesColumn(expr.value, key);
    case "in":
    case "not-in":
      return (
        refReferencesColumn(expr.value, key) ||
        expr.list.some((item) => refReferencesColumn(item, key))
      );
    default:
      return (
        refReferencesColumn(expr.left, key) ||
        refReferencesColumn(expr.right, key)
      );
  }
}

function refReferencesColumn(ref: ValueRef, key: string): boolean {
  if ("literal" in ref) return false;
  return ref.ref === `row.${key}` || ref.ref.startsWith(`row.${key}.`);
}
```

```typescript
import { Injectable, BadRequestException, ConflictException } from "@nestjs/common";
import { AuditService } from "@/audit/audit.service";
import { TablesDbService } from "./tables-db.service";
import { validateColumnDefs, buildRowZodSchema } from "./column-validation";
import { validateLookupDefs } from "./lookup-validation";
import { findLookupsReferencingColumn } from "./dependency-check";
import type { ColumnDef, LookupDef } from "./types";

export interface CreateTableArgs {
  actor_id: string;
  group_id: string;
  table_id: string;
  label: string;
  description: string | null;
}

@Injectable()
export class TablesService {
  constructor(
    private readonly db: TablesDbService,
    private readonly audit: AuditService,
  ) {}

  async createTable(args: CreateTableArgs) {
    const created = await this.db.createTable({
      group_id: args.group_id,
      table_id: args.table_id,
      label: args.label,
      description: args.description,
    });
    await this.audit.recordEvent({
      event_type: "tables.created",
      resource_type: "table",
      resource_id: created.id,
      actor_id: args.actor_id,
      group_id: args.group_id,
      payload: { table_id: args.table_id, label: args.label },
    });
    return created;
  }

  async listTables(group_id: string) {
    return this.db.listTables(group_id);
  }

  async getTable(group_id: string, table_id: string) {
    return this.db.findTable(group_id, table_id);
  }

  async updateTableMetadata(
    actor_id: string,
    group_id: string,
    table_id: string,
    patch: { label?: string; description?: string | null },
  ) {
    const updated = await this.db.updateTableMetadata(group_id, table_id, patch);
    await this.audit.recordEvent({
      event_type: "tables.updated",
      resource_type: "table",
      resource_id: updated.id,
      actor_id,
      group_id,
      payload: patch,
    });
    return updated;
  }

  async deleteTable(actor_id: string, group_id: string, table_id: string) {
    const existing = await this.db.findTable(group_id, table_id);
    if (!existing) return;
    await this.db.deleteTable(group_id, table_id);
    await this.audit.recordEvent({
      event_type: "tables.deleted",
      resource_type: "table",
      resource_id: existing.id,
      actor_id,
      group_id,
      payload: { table_id },
    });
  }

  async addColumn(actor_id: string, group_id: string, table_id: string, col: ColumnDef) {
    const t = await this.db.findTable(group_id, table_id);
    if (!t) throw new BadRequestException("table not found");
    const next = [...((t.columns as ColumnDef[]) ?? []), col];
    try {
      validateColumnDefs(next);
      validateLookupDefs((t.lookups as LookupDef[]) ?? [], next);
    } catch (e) {
      throw new BadRequestException((e as Error).message);
    }
    const updated = await this.db.addColumn(group_id, table_id, col);
    await this.audit.recordEvent({
      event_type: "tables.column.added",
      resource_type: "table",
      resource_id: updated.id,
      actor_id,
      group_id,
      payload: { column: col },
    });
    return updated;
  }

  async updateColumn(
    actor_id: string,
    group_id: string,
    table_id: string,
    key: string,
    next: ColumnDef,
  ) {
    const t = await this.db.findTable(group_id, table_id);
    if (!t) throw new BadRequestException("table not found");
    const cols = ((t.columns as ColumnDef[]) ?? []).map((c) => (c.key === key ? next : c));
    try {
      validateColumnDefs(cols);
      validateLookupDefs((t.lookups as LookupDef[]) ?? [], cols);
    } catch (e) {
      throw new BadRequestException((e as Error).message);
    }
    const updated = await this.db.updateColumn(group_id, table_id, key, next);
    await this.audit.recordEvent({
      event_type: "tables.column.updated",
      resource_type: "table",
      resource_id: updated.id,
      actor_id,
      group_id,
      payload: { column_key: key, before: (t.columns as ColumnDef[]).find((c) => c.key === key), after: next },
    });
    return updated;
  }

  async removeColumn(actor_id: string, group_id: string, table_id: string, key: string) {
    const t = await this.db.findTable(group_id, table_id);
    if (!t) throw new BadRequestException("table not found");
    const dependents = findLookupsReferencingColumn(
      (t.lookups as LookupDef[]) ?? [],
      key,
    );
    if (dependents.length > 0) {
      throw new ConflictException(
        `column "${key}" is referenced by lookups: ${dependents.join(", ")}`,
      );
    }
    const updated = await this.db.removeColumn(group_id, table_id, key);
    await this.audit.recordEvent({
      event_type: "tables.column.removed",
      resource_type: "table",
      resource_id: updated.id,
      actor_id,
      group_id,
      payload: { column_key: key },
    });
    return updated;
  }

  async addLookup(
    actor_id: string,
    group_id: string,
    table_id: string,
    lookup: LookupDef,
  ) {
    const t = await this.db.findTable(group_id, table_id);
    if (!t) throw new BadRequestException("table not found");
    const next = [...((t.lookups as LookupDef[]) ?? []), lookup];
    try {
      validateLookupDefs(next, (t.columns as ColumnDef[]) ?? []);
    } catch (e) {
      throw new BadRequestException((e as Error).message);
    }
    const updated = await this.db.addLookup(group_id, table_id, lookup);
    await this.audit.recordEvent({
      event_type: "tables.lookup.added",
      resource_type: "table",
      resource_id: updated.id,
      actor_id,
      group_id,
      payload: { lookup },
    });
    return updated;
  }

  async updateLookup(
    actor_id: string,
    group_id: string,
    table_id: string,
    name: string,
    next: LookupDef,
  ) {
    const t = await this.db.findTable(group_id, table_id);
    if (!t) throw new BadRequestException("table not found");
    const lookups = ((t.lookups as LookupDef[]) ?? []).map((l) => (l.name === name ? next : l));
    try {
      validateLookupDefs(lookups, (t.columns as ColumnDef[]) ?? []);
    } catch (e) {
      throw new BadRequestException((e as Error).message);
    }
    const updated = await this.db.updateLookup(group_id, table_id, name, next);
    await this.audit.recordEvent({
      event_type: "tables.lookup.updated",
      resource_type: "table",
      resource_id: updated.id,
      actor_id,
      group_id,
      payload: { lookup_name: name, after: next },
    });
    return updated;
  }

  async removeLookup(actor_id: string, group_id: string, table_id: string, name: string) {
    const updated = await this.db.removeLookup(group_id, table_id, name);
    await this.audit.recordEvent({
      event_type: "tables.lookup.removed",
      resource_type: "table",
      resource_id: updated.id,
      actor_id,
      group_id,
      payload: { lookup_name: name },
    });
    return updated;
  }

  async createRow(actor_id: string, group_id: string, table_id: string, data: Record<string, unknown>) {
    const t = await this.db.findTable(group_id, table_id);
    if (!t) throw new BadRequestException("table not found");
    const schema = buildRowZodSchema((t.columns as ColumnDef[]) ?? []);
    const parsed = schema.parse(data) as Record<string, unknown>;
    const row = await this.db.createRow(group_id, table_id, parsed);
    await this.audit.recordEvent({
      event_type: "tables.row.created",
      resource_type: "table_row",
      resource_id: row.id,
      actor_id,
      group_id,
      payload: { table_id, after: parsed },
    });
    return row;
  }

  async listRows(group_id: string, table_id: string, opts: { offset: number; limit: number }) {
    return this.db.listRows(group_id, table_id, opts);
  }

  async getRow(group_id: string, table_id: string, id: string) {
    return this.db.findRow(group_id, table_id, id);
  }

  async updateRow(
    actor_id: string,
    group_id: string,
    table_id: string,
    id: string,
    input: { data: Record<string, unknown>; expected_updated_at: Date },
  ) {
    const t = await this.db.findTable(group_id, table_id);
    if (!t) throw new BadRequestException("table not found");
    const schema = buildRowZodSchema((t.columns as ColumnDef[]) ?? []);
    const parsed = schema.parse(input.data) as Record<string, unknown>;
    const before = await this.db.findRow(group_id, table_id, id);
    let row;
    try {
      row = await this.db.updateRow(group_id, table_id, id, {
        data: parsed,
        expected_updated_at: input.expected_updated_at,
      });
    } catch (e) {
      throw new ConflictException((e as Error).message);
    }
    await this.audit.recordEvent({
      event_type: "tables.row.updated",
      resource_type: "table_row",
      resource_id: row.id,
      actor_id,
      group_id,
      payload: { table_id, before: before?.data, after: parsed },
    });
    return row;
  }

  async deleteRow(actor_id: string, group_id: string, table_id: string, id: string) {
    const before = await this.db.findRow(group_id, table_id, id);
    if (!before) return;
    await this.db.deleteRow(group_id, table_id, id);
    await this.audit.recordEvent({
      event_type: "tables.row.deleted",
      resource_type: "table_row",
      resource_id: id,
      actor_id,
      group_id,
      payload: { table_id, before: before.data },
    });
  }
}
```

```typescript
// apps/temporal/src/tables/lookup-engine.ts
import { evaluateConditionWithBindings } from "../expression-evaluator";
import type { LookupDef } from "./types";

export class LookupError extends Error {
  constructor(public code: string, message: string) {
    super(message);
  }
}

export function executeLookup(
  lookup: LookupDef,
  params: Record<string, unknown>,
  rows: Array<Record<string, unknown>>,
): Record<string, unknown> | Array<Record<string, unknown>> | null {
  const matched = rows.filter((row) =>
    evaluateConditionWithBindings(lookup.filter, { ctx: {}, param: params, row }),
  );
  if (lookup.order && lookup.order.length > 0) {
    matched.sort((a, b) => {
      for (const o of lookup.order!) {
        const av = a[o.field];
        const bv = b[o.field];
        if (av === bv) continue;
        const cmp = (av as never) < (bv as never) ? -1 : 1;
        return o.direction === "asc" ? cmp : -cmp;
      }
      return 0;
    });
  }
  switch (lookup.pick) {
    case "all":
      return matched;
    case "first":
      return matched[0] ?? null;
    case "last":
      return matched[matched.length - 1] ?? null;
    case "one":
      if (matched.length === 0) {
        throw new LookupError("TABLES_NO_MATCH", `no rows matched lookup ${lookup.name}`);
      }
      if (matched.length > 1) {
        throw new LookupError(
          "TABLES_AMBIGUOUS_MATCH",
          `${matched.length} rows matched lookup ${lookup.name} (expected exactly 1)`,
        );
      }
      return matched[0];
  }
}
```

```typescript
// apps/temporal/src/activities/tables-lookup.ts
import { ApplicationFailure } from "@temporalio/common";
import { executeLookup, LookupError } from "../tables/lookup-engine";
import type { LookupDef } from "../tables/types";
// Import prisma client per the existing convention used by other activities.
// Example (verify against repository):
import { prisma } from "../prisma";

export class TablesLookupError extends Error {}

export interface TablesLookupInput {
  groupId: string;
  tableId: string;
  lookupName: string;
  [paramName: string]: unknown;
}

export interface TablesLookupOutput {
  result: Record<string, unknown> | Array<Record<string, unknown>> | null;
}

export async function tablesLookup(input: TablesLookupInput): Promise<TablesLookupOutput> {
  const { groupId, tableId, lookupName, ...params } = input;
  const table = await prisma().table.findUnique({
    where: { group_id_table_id: { group_id: groupId, table_id: tableId } },
  });
  if (!table) {
    throw ApplicationFailure.create({
      type: "TABLES_NOT_FOUND",
      message: `table not found: groupId=${groupId} tableId=${tableId}`,
    });
  }
  const lookups = (table.lookups as LookupDef[]) ?? [];
  const lookup = lookups.find((l) => l.name === lookupName);
  if (!lookup) {
    throw ApplicationFailure.create({
      type: "TABLES_LOOKUP_NOT_FOUND",
      message: `lookup not found: ${lookupName}`,
    });
  }
  const rows = await prisma().tableRow.findMany({
    where: { group_id: groupId, table_id: tableId },
  });
  try {
    const result = executeLookup(
      lookup,
      params,
      rows.map((r) => r.data as Record<string, unknown>),
    );
    return { result };
  } catch (e) {
    if (e instanceof LookupError) {
      throw ApplicationFailure.create({ type: e.code, message: e.message });
    }
    throw e;
  }
}
```

```typescript
// apps/frontend/src/features/tables/utils/build-row-zod-schema.ts
import { z } from "zod";
import type { ColumnDef } from "../types";

export function buildRowZodSchema(cols: ColumnDef[]): z.ZodTypeAny {
  const shape: Record<string, z.ZodTypeAny> = {};
  for (const c of cols) {
    let base: z.ZodTypeAny;
    switch (c.type) {
      case "string": base = z.string(); break;
      case "number": base = z.number(); break;
      case "boolean": base = z.boolean(); break;
      case "date": base = z.string().regex(/^\d{4}-\d{2}-\d{2}$/); break;
      case "datetime": base = z.string().datetime({ offset: true }); break;
      case "enum": base = z.enum(c.enumValues as [string, ...string[]]); break;
    }
    shape[c.key] = c.required ? base! : base!.optional();
  }
  return z.object(shape).strip();
}
```

```typescript
// apps/frontend/src/features/tables/lookup-templates/earliest-after.ts
import { Select, TextInput, Stack } from "@mantine/core";
import type { LookupTemplate } from "./types";

export const earliestAfter: LookupTemplate = {
  id: "earliest-after",
  label: "Earliest after / on",
  toLookupDef(name, v, columns) {
    const col = String(v.column);
    const param = String(v.param);
    const colDef = columns.find((c) => c.key === col);
    return {
      name,
      params: [{ name: param, type: colDef?.type ?? "string" }],
      filter: { operator: "lte", left: { ref: `param.${param}` }, right: { ref: `row.${col}` } },
      order: [{ field: col, direction: "asc" }],
      pick: "first",
      templateId: "earliest-after",
      templateConfig: { column: col, param },
    };
  },
  fromLookupDef(l) {
    if (l.templateId === "earliest-after" && l.templateConfig) return l.templateConfig as Record<string, unknown>;
    return null;
  },
  renderFields({ columns, values, setValue }) {
    return (
      <Stack>
        <Select
          label="Column"
          required
          data={columns.map((c) => ({ value: c.key, label: c.label }))}
          value={(values.column as string) ?? null}
          onChange={(v) => setValue("column", v)}
        />
        <TextInput
          label="Param name"
          required
          value={(values.param as string) ?? ""}
          onChange={(e) => setValue("param", e.currentTarget.value)}
        />
      </Stack>
    );
  },
};
```

Apply the same pattern for:
- **exact-match**: filter `equals`, `pick: "one"`, single column + param.
- **range-contains**: filter `and([gte, lte])` with two columns, `pick: "one"`.
- **latest-before**: filter `lte` (param ≥ row.column), order `desc`, `pick: "first"`. Wait — clarify direction: `column ≤ param` means filter is `{ operator: "lte", left: { ref: "row.col" }, right: { ref: "param.x" } }`, order `desc`, `pick: "first"`.
- **multi-field-exact**: list of (column, param) pairs, joined with `and` of `equals`, `pick: "one"`.
- **custom-json**: a JSON editor textarea for `filter`, plus list editors for `params`/`order` and a Select for `pick`. `fromLookupDef` always returns the canonical shape (it's the catch-all).

```typescript
// apps/frontend/src/features/tables/lookup-templates/index.ts
import { exactMatch } from "./exact-match";
import { rangeContains } from "./range-contains";
import { latestBefore } from "./latest-before";
import { earliestAfter } from "./earliest-after";
import { multiFieldExact } from "./multi-field-exact";
import { customJson } from "./custom-json";
import type { LookupDef } from "../types";
import type { LookupTemplate } from "./types";

export const LOOKUP_TEMPLATES: LookupTemplate[] = [
  exactMatch, rangeContains, latestBefore, earliestAfter, multiFieldExact, customJson,
];

export function templateFor(lookup: LookupDef): LookupTemplate {
  // Try templateId hint first
  if (lookup.templateId) {
    const t = LOOKUP_TEMPLATES.find((x) => x.id === lookup.templateId);
    if (t && t.fromLookupDef(lookup)) return t;
  }
  // Probe all templates for round-trip match
  for (const t of LOOKUP_TEMPLATES) {
    if (t.id !== "custom-json" && t.fromLookupDef(lookup)) return t;
  }
  return customJson;
}
```
